 17 /**
 18 *		@class EventHandler
 19 *
 20 *		@constructor
 21 *		@param {document}	document document object
 22 *		@param {window} window window object
 23 *		@param {console} console console object
 24 **/
 25 
 26 var EventHandler = (function(window, document, console) {
 27   var regElement = {};
 28   var broadcastElement = {};
 29   var logging = {};
 30   var minPrio = 0;
 31   var maxPrio = 3;
 32   var isBlock = false;
 33   var eventList = ["CLICK","DBLCLICK", "KEYDOWN", "KEYUP", "KEYPRESS", "DRAGDROP", "FOCUS", "BLUR", "CHANGE", "SCROLL",
 34     "DISABLED", "ENABLED", "SHOW", "HIDDEN", "SETERROR", "UNSETERROR", "SELECTTEXT", "DESELECTTEXT"];
 35 
 36 
 37   // register for all events in eventList the EventHandler check routine, call checkListener
 38   eventList.forEach(function(ev) {
 39     document.body.addEventListener(ev.toLowerCase(), function(e) {
 40       var src = e.srcElement || e.target;
 41       if (e.type.toLowerCase() === 'click') {
 42         //console.warn(src.id);
 43       }
 44       if (!src.id) {
 45         var temp = HTMLHelper.getParentId(src);
 46         if (temp) {
 47           src.id = HTMLHelper.getParentId(temp);
 48         }
 49       }
 50 
 51       checkListener(e, false);
 52       checkListener(e, true);
 53     }, true);
 54   });
 55 
 56 
 57   /** register new event if is not existing
 58   *		@function registerNewEvent
 59   *		@param {string} event event name
 60   *
 61   **/
 62   var registerNewEvent = function(event) {
 63     if (event && !existsEvent(event)) {
 64       eventList.push(event);
 65     } else {
 66       console.warn('cannot add new event: '+event);
 67     }
 68   };
 69 
 70 
 71   var existsEvent = function(event) {
 72     var found = false;
 73     if (event) {
 74       var k = 0;
 75       while (!found && k < eventList.length) {
 76         found = (eventList[k].toLowerCase() === event.toLowerCase());
 77         k++;
 78       }
 79     }
 80 
 81     return found;
 82   };
 83 
 84 
 85   /** cancel event propagation
 86   *		@function stopEvent
 87   *		@param {string} event event name
 88   *
 89   **/
 90   var stopEvent = function(event) {
 91     event = event || window.event; // cross-browser event
 92     if (event.stopPropagation) {
 93       // W3C standard variant
 94       event.stopPropagation();
 95     } else {
 96       // IE variant
 97       event.cancelBubble = true;
 98     }
 99 
100     if (event.preventDefault) {
101       event.preventDefault();
102     }
103   };
104 
105 
106   var checkEvent = function(event) {
107     var result =  (event && ((event.target && (event.target.id || event.target.tagName)) || event.tagName) && event.type);
108     return result;
109   };
110 
111 
112   var existsHideClass = function(elem) {
113     var isHidden = false;
114     if (elem) {
115       var cl = elem.classList;
116       if (cl) {
117         isHidden = (cl.contains('ng-hide'));
118       }
119       elem = elem.parentNode;
120 
121       if (isHidden) {
122         console.warn('found hidden class!');
123       }
124     }
125 
126   };
127 
128 
129   var executeCallback = function(nArr, e) {
130     if (nArr.EventCallback) {
131       if (Array.isArray(nArr.EventCallback)) {
132         for (var prio=minPrio; prio <= maxPrio; prio++) {
133           for (var j=0; j < nArr.EventCallback.length; j++) {
134             //console.log('execute '+nArr.EventCallback[j]);
135             //console.warn('execute '+nArr.id+' prio: '+nArr.prio);
136             //console.log('execute '+elemId+tevent);
137             if (nArr.prio === prio && typeof nArr.EventCallback[j] === 'function') {
138               nArr.EventCallback[j](e);
139             }
140           }
141         }
142       } else {
143         if (typeof nArr.EventCallback === 'function') {
144           nArr.EventCallback(e);
145         }
146       }
147 
148     } else {
149       console.warn('found no entry for event: '+nArr.id);
150     }
151   };
152 
153 
154   // {boolean} byId - checking for id or for element tag
155   var checkListener = function(e, byId) {
156     var found = false;
157 
158     if (checkEvent(e)) {
159       var elem = e.target || e.srcElement;
160       var elemId = '';
161       if (!byId) {
162         elemId = (elem.tagName.toLowerCase() || e.tagName.toLowerCase());
163       } else {
164         elemId = elem.id;
165       }
166 
167       if (elemId && !existsHideClass(elem)) {
168         var tevent = e.type.toLowerCase();
169         var tid = regElement[elemId + tevent];
170         if (logging[elemId]) {
171           console.log('call callback id: '+elemId);
172         }
173 
174         if (tid && tid.length > 0) {
175           for (var i=0; i < tid.length; i++) {
176             var nArr = tid[i];
177             if (nArr.id === elemId && nArr.type && nArr.type.toLowerCase() === tevent) {
178               found = true;
179               executeCallback(nArr, e);
180             }
181           }
182         }
183       }
184     }
185 
186     return found;
187   };

190   /** register callback for element id and event type
191   *		@function registerElementById
192   *		@param {string} id element id
193   *   @param {string} type event name
194   *   @param {function} EventCallback callback function
195   *   @param {number} prio priority
196   *   @param {string} section section name
197   *   @param {boolean} only executing only this callback
198   *
199   **/
200   var registerElementById = function(id, type, EventCallback, prio, section, only) {
201     if (id && type) {
202       type = type.toLowerCase();
203       if (existsEvent(type)) {
204         var nId = '' + id + '' + type;
205 
206         if (!prio || Number.isInteger(prio)) {
207           prio = minPrio;
208         }
209         if (regElement[nId] !== null && regElement[nId] !== undefined && regElement[nId].length > 0) {
210         } else {
211           regElement[nId] = [];
212         }
213 
214         if (!section) {
215           section = '';
216         }
217 
218         if (only !== undefined && only !== null && only) {
219           regElement[nId][0] = {id: id, EventCallback : EventCallback, type : type, prio : prio, section : section};
220         } else {
221           regElement[nId].push({id: id, EventCallback : EventCallback, type : type, prio : prio, section : section});
222         }
223       }
224     } else {
225     }
226 
227   };

230   /** delete all callbacks for element id
231   *		@function deleteAllElementsByID
232   *		@param {string} id element id
233   *
234   **/
235   var deleteAllElementsByID = function(id) {
236     if (id) {
237       for (var key in regElement) {
238         var found = false;
239         var k = 0;
240         while (!found && k < regElement[key].length) {
241           found = (regElement[key][k].id && regElement[key][k].id.toLowerCase() === id.toLowerCase());
242           k++;
243         }
244         if (found) {
245           regElement[key] = [];
246         }
247       }
248     }
249   };
250 
251 
252   /** delete all callbacks for section
253   *		@function deleteAllElementsBySection
254   *		@param {string} section section name
255   *
256   **/
257   var deleteAllElementsBySection = function(section) {
258     // hier werden alle sections gelscht
259     if (section) {
260       var deleteID = [];
261       for (var key in regElement) {
262         var found = false;
263         var k = 0;
264         while (!found && k < regElement[key].length) {
265           found = (regElement[key][k].section && regElement[key][k].section.toLowerCase() === section.toLowerCase());
266           k++;
267         }
268         if (found) {
269           if (regElement[key][0] && regElement[key][0].id) {
270             var id = regElement[key][0].id;
271             if (deleteID.indexOf(id) < 0) {
272               this.deleteAllElementsByID(id);
273               this.deleteAllElementsByID(id + 'Arrow');
274               deleteID.push(id);
275             }
276           }
277           regElement[key] = [];
278         }
279       }
280     }
281   };

284   /** register callback for element and event type
285   *		@function registerElement
286   *		@param {string} element element, tag name
287   *   @param {string} type event name
288   *   @param {function} EventCallback callback function
289   *   @param {number} prio priority
290   *
291   **/
292   var registerElement = function(element, type, EventCallback, prio) {
293     if (element && type) {
294       type = type.toLowerCase();
295       if (existsEvent(type)) {
296         var nId = '' + element.toLowerCase() + '' + type;
297 
298         if (!prio) {
299           prio = minPrio;
300         }
301         if (!regElement[nId]) {
302           regElement[nId] = [];
303           regElement[nId][0] = {id: element, EventCallback : EventCallback, type : type, prio : prio};
304         }
305 
306       }
307     } else {
308       console.warn('cannot add event: '+type+' to id: '+element);
309     }
310   };
311

313   /** trigger event for element id and event type
314   *		@function triggerEvent
315   *		@param {string} ids elements id
316   *   @param {string} type event name
317   *
318   **/
319   var triggerEvent = function(ids, type) {
320     if (type) {
321       type = type.toLowerCase();
322       if (existsEvent(type) && ids && Array.isArray(ids)) {
323         for (var i=0; i < ids.length; i++) {
324           var id = ids[i];
325           var e = {
326             target : {id: id, tagName : ''},
327             type: type,
328           };
329           checkListener(e, true);
330         }
331       }
332     }
333   };
334 
335 
336   /** trigger broadcast
337   *		@function triggerBroadcast
338   *   @param {string} type event name
339   *
340   **/
341   var triggerBroadcast = function(type) {
342     if (type && existsEvent(type)) {
343       var tid = broadcastElement[type];
344       if (tid) {
345         for (var i = 0; i < tid.length; i++) {
346           executeCallback(tid[i], null);
347         }
348       }
349     }
350   };
351 
352 
353   /** register callback for broadcast
354   *		@function registerBroadcast
355   *		@param {string} id element id
356   *   @param {string} type event name
357   *   @param {function} EventCallback callback function
358   *
359   **/
360   var registerBroadcast = function(id, type, EventCallback) {
361     if (type && existsEvent(type)) {
362       if (!broadcastElement[type]) {
363         broadcastElement[type] = [];
364       }
365       broadcastElement[type].push({id : id, type : type, EventCallback : EventCallback});
366     }
367   };
368 
369 
370 
371 
372   return {
373     registerElementById : registerElementById,
374     registerElement : registerElement,
375     triggerEvent : triggerEvent,
376     triggerBroadcast : triggerBroadcast,
377     registerBroadcast : registerBroadcast,
378     registerNewEvent : registerNewEvent,
379     stopEvent : stopEvent,
380     deleteAllElementsByID : deleteAllElementsByID,
381     deleteAllElementsBySection : deleteAllElementsBySection
382   };
383 
384 }(window, document, console));
